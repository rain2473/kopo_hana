package game;

import java.util.Random;
import java.lang.Math;

public class Cleric {
    static final int HP_MAX = 50;
    static final int MP_MAX = 10;

    String name;

    int hp = Cleric.HP_MAX;
    int mp = Cleric.MP_MAX;

    Random random = new Random();

    Cleric(String name, int hp, int mp) {
        this.name = name;
        this.hp = hp;
        this.mp = mp;
    }

    Cleric(String name, int hp) {
        this(name, hp, Cleric.MP_MAX);
    }

    Cleric(String name) {
        this(name, Cleric.HP_MAX, Cleric.MP_MAX);
    }


    void selfAid() {
        if (mp >= 5) {
            this.hp = Cleric.HP_MAX;
            this.mp -= 5;
        }
    }

    int pray(int sec) {
        int mpRecovery = random.nextInt(3) + sec;
        this.mp += mpRecovery;
        this.mp = Math.min(this.mp, Cleric.MP_MAX);
        return mpRecovery;
    }
}
