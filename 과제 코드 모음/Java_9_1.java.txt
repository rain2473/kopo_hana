package game;

import java.util.Random;
import java.lang.Math;

public class Cleric {
    static final int HP_MAX = 50;
    static final int MP_MAX = 10;
    
    String name;

    int hp = Cleric.HP_MAX;
    int mp = Cleric.MP_MAX;

    Random random = new Random();

    void selfAid() {
        if (mp >= 5) {
            this.hp = Cleric.HP_MAX;
            this.mp -= 5;
        }
    }

    int pray(int sec) {
        int mpRecovery = random.nextInt(3) + sec;
        this.mp += mpRecovery;
        this.mp = Math.min(this.mp, Cleric.MP_MAX);
        return mpRecovery;
    }
}
