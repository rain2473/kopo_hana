package com.example;

import java.util.*;
import java.lang.*;

public class Main {
    public static void main(String[] args) {
        showOutput(converte(generate(getInputs())));
    }

    public static int[] getInputs() {
        Scanner scanner = new Scanner(System.in);
        String[] inputs = scanner.nextLine().split(" ");
        scanner.close();
        int count = inputs.length;
        int[] result = new int[count];
        for (int i = 0; i < count; i++) {
            result[i] = Integer.parseInt(inputs[i]);
        }
        return result;
    }

    public static int[] generate(int[] inputs, int count) {
        int[] result = new int[count];
        int d = inputs[1];
        result[0] = inputs[0];
        for (int i = 1; i < count; i++) {
            result[i] = result[i - 1] + d;
        }
        return result;
    }

    public static int[] generate(int[] inputs) {
        return generate(inputs, 10);
    }

    public static String converte(int[] inputs) {
        StringBuilder result = new StringBuilder("");
        int count = inputs.length;
        for (int i = 0; i < count; i++) {
            result.append(Integer.toString(inputs[i]));
            if (i != count - 1) {
                result.append(" ");
            } else {
                result.append("\n");
            }
        }
        return result.toString();
    }

    public static void showOutput(String input) {
        System.out.println(input);
    }
}
